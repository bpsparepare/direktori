```sql
    kd_prov,
    kd_kab,
    kd_kec,
    kd_desa,
    COUNT(*) as jumlah_usaha
FROM direktori 
WHERE sumber_data = 'BPS'
GROUP BY kd_prov, kd_kab, kd_kec, kd_desa
ORDER BY kd_prov, kd_kab, kd_kec, kd_desa;

SELECT 
    id_sbr,
    nama_usaha,
    alamat,
    CONCAT(kd_prov, kd_kab, kd_kec, kd_desa) as kode_wilayah,
    nomor_telepon,
    email,
    created_at
FROM direktori 
WHERE sumber_data = 'BPS'
ORDER BY created_at DESC
LIMIT 10;
```
